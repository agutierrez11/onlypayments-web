import { NextResponse } from 'next/server';import { db } from '@/lib/db';import { paymentMethods } from '@/lib/db/schema';import { eq, and, like } from 'drizzle-orm';export const runtime = 'edge';export async function GET(request: Request) {  const { searchParams } = new URL(request.url);  const country = searchParams.get('country');  const search = searchParams.get('search');  const sortBy = searchParams.get('sortBy') || 'volume';  let query = db.select().from(paymentMethods).where(eq(paymentMethods.isActive, true));  if (country && country !== 'Todos') {    query = query.where(eq(paymentMethods.countryName, country));  }  if (search) {    query = query.where(      like(paymentMethods.name, `%${search}%`)    );  }  const results = await query;  // Sort  const sorted = results.sort((a, b) => {    if (sortBy === 'volume') return Number(b.monthlyVolumeUsd) - Number(a.monthlyVolumeUsd);    if (sortBy === 'fee') return Number(a.feePercentage) - Number(b.feePercentage);    return a.name.localeCompare(b.name);  });  return NextResponse.json(sorted, {    headers: {      'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=300',    },  });}

📋 Checklist de honestidad antes de lanzar
☐ Ningún logo de empresa sin permiso escrito
☐ Ninguna cifra sin fuente citada
☐ Ningún testimonio sin autor real verificable
☐ Ningún claim de “mejor/más grande/#1” sin evidencia
☐ Fechas de actualización visibles en datos
☐ Disclaimer de “datos estimados” donde aplique
☐ Link a fuentes de datos públicos
☐ Formulario de “reportar error” visible
